103std::optional<T> FormData::get_as(std::string_view name) const {
104 auto value = get(name);
105 if (!value) return std::nullopt;
106
107 if constexpr (std::is_same_v<T, std::string>) {
108 return std::string(*value);
109 } else if constexpr (std::is_same_v<T, int>) {
110 try { return std::stoi(std::string(*value)); }
111 catch (...) { return std::nullopt; }
112 } else if constexpr (std::is_same_v<T, int64_t>) {
113 try { return std::stoll(std::string(*value)); }
114 catch (...) { return std::nullopt; }
115 } else if constexpr (std::is_same_v<T, double>) {
116 try { return std::stod(std::string(*value)); }
117 catch (...) { return std::nullopt; }
118 } else if constexpr (std::is_same_v<T, bool>) {
119 return *value == "true" || *value == "1" || *value == "on";
120 } else {
121 static_assert(sizeof(T) == 0, "Unsupported type for get_as");
122 }
