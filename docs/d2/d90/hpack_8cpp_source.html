47expected<std::vector<uint8_t>, Error> HpackEncoder::encode(std::span<const Header> headers) {
48 if (!deflater_) {
49 return unexpected(Error::io(IoError::Unknown, "HPACK encoder not initialized"));
50 }
51
52 // Convert to nghttp2 format
53 std::vector<nghttp2_nv> nvs;
54 nvs.reserve(headers.size());
55
56 for (const auto& h : headers) {
57 nghttp2_nv nv;
58 nv.name = reinterpret_cast<uint8_t*>(const_cast<char*>(h.name.data()));
59 nv.namelen = h.name.size();
60 nv.value = reinterpret_cast<uint8_t*>(const_cast<char*>(h.value.data()));
61 nv.valuelen = h.value.size();
62 nv.flags = NGHTTP2_NV_FLAG_NONE;
63 nvs.push_back(nv);
64 }
65
66 // Calculate upper bound for encoded size
67 size_t buflen = nghttp2_hd_deflate_bound(deflater_, nvs.data(), nvs.size());
68
69 std::vector<uint8_t> result(buflen);
70
71 // Encode
72 ssize_t rv = nghttp2_hd_deflate_hd(
73 deflater_,
74 result.data(),
75 result.size(),
76 nvs.data(),
77 nvs.size()
78 );
79
80 if (rv < 0) {
81 return unexpected(Error::io(IoError::Unknown,
82 std::string("HPACK encode error: ") + nghttp2_strerror(static_cast<int>(rv))));
83 }
84
85 result.resize(static_cast<size_t>(rv));
86 return result;
87}

133expected<std::vector<Header>, Error> HpackDecoder::decode(std::span<const uint8_t> data) {
134 if (!inflater_) {
135 return unexpected(Error::io(IoError::Unknown, "HPACK decoder not initialized"));
136 }
137
138 std::vector<Header> result;
139
140 const uint8_t* in = data.data();
141 size_t inlen = data.size();
142
143 while (inlen > 0) {
144 nghttp2_nv nv;
145 int inflate_flags = 0;
146
147 ssize_t rv = nghttp2_hd_inflate_hd2(
148 inflater_,
149 &nv,
150 &inflate_flags,
151 in,
152 inlen,
153 1 // in_final - we're processing complete header block
154 );
155
156 if (rv < 0) {
157 return unexpected(Error::io(IoError::Unknown,
158 std::string("HPACK decode error: ") + nghttp2_strerror(static_cast<int>(rv))));
159 }
160
161 in += rv;
162 inlen -= static_cast<size_t>(rv);
163
164 if (inflate_flags & NGHTTP2_HD_INFLATE_EMIT) {
165 result.push_back({
166 std::string(reinterpret_cast<const char*>(nv.name), nv.namelen),
167 std::string(reinterpret_cast<const char*>(nv.value), nv.valuelen)
168 });
169 }
170
171 if (inflate_flags & NGHTTP2_HD_INFLATE_FINAL) {
172 break;
173 }
174 }
175
176 // End header block
177 nghttp2_hd_inflate_end_headers(inflater_);
178
179 return result;
180}

199const Header* find_header(std::span<const Header> headers, std::string_view name) {
200 for (const auto& h : headers) {
201 // Pseudo-headers are case-sensitive, regular headers are case-insensitive
202 if (name[0] == ':') {
203 if (h.name == name) return &h;
204 } else {
205 if (h.name.size() == name.size()) {
206 bool match = true;
207 for (size_t i = 0; i < name.size(); ++i) {
208 if (std::tolower(static_cast<unsigned char>(h.name[i])) !=
209 std::tolower(static_cast<unsigned char>(name[i]))) {
210 match = false;
211 break;
212 }
213 }
214 if (match) return &h;
215 }
216 }
217 }
218 return nullptr;
219}

246bool validate_request_headers(std::span<const Header> headers) {
247 // Must have :method, :scheme, :path
248 // :authority is optional but recommended
249 bool has_method = false;
250 bool has_scheme = false;
251 bool has_path = false;
252 bool pseudo_ended = false;
253
254 for (const auto& h : headers) {
255 if (h.is_pseudo()) {
256 if (pseudo_ended) {
257 // Pseudo-headers must come before regular headers
258 return false;
259 }
260 if (h.name == ":method") has_method = true;
261 else if (h.name == ":scheme") has_scheme = true;
262 else if (h.name == ":path") has_path = true;
263 else if (h.name == ":authority") { /* optional */ }
264 else if (h.name == ":status") {
265 // :status is for responses only
266 return false;
267 }
268 } else {
269 pseudo_ended = true;
270 // Header names must be lowercase
271 for (char c : h.name) {
272 if (std::isupper(static_cast<unsigned char>(c))) {
273 return false;
274 }
275 }
276 }
277 }
278
279 return has_method && has_scheme && has_path;
280}

282bool validate_response_headers(std::span<const Header> headers) {
283 // Must have :status
