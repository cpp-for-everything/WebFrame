29class ConnectionPool {
30public:
31 using ConnectionFactory = std::function<std::unique_ptr<Connection>()>;
32 using ConnectionResetter = std::function<void(Connection&)>;
33
34private:
35 std::vector<std::unique_ptr<Connection>> pool_;
36 mutable std::mutex mutex_;
37 ConnectionPoolConfig config_;
38 ConnectionResetter resetter_;
39
40 std::atomic<size_t> active_count_{0};
41 std::atomic<size_t> total_acquired_{0};
42 std::atomic<size_t> total_reused_{0};
43
44public:
45 explicit ConnectionPool(ConnectionPoolConfig config = {})
46 : config_(std::move(config))
47 {
48 pool_.reserve(config_.max_size);
49 }
50
51 // Set custom reset function (called when connection is returned)
52 void set_resetter(ConnectionResetter resetter) {
53 resetter_ = std::move(resetter);
54 }
55
56 // Acquire a connection from the pool (or nullptr if empty)
57 std::unique_ptr<Connection> acquire() {
58 std::lock_guard<std::mutex> lock(mutex_);
59 total_acquired_.fetch_add(1, std::memory_order_relaxed);
60
61 if (pool_.empty()) {
62 return nullptr;
63 }
64
65 total_reused_.fetch_add(1, std::memory_order_relaxed);
66 auto conn = std::move(pool_.back());
67 pool_.pop_back();
68 active_count_.fetch_add(1, std::memory_order_relaxed);
69 return conn;
70 }
71
72 // Release a connection back to the pool
73 void release(std::unique_ptr<Connection> conn) {
74 if (!conn || !config_.enable_reuse) {
75 return;
76 }
77
78 active_count_.fetch_sub(1, std::memory_order_relaxed);
79
80 // Reset connection state if resetter is set
81 if (resetter_) {
82 resetter_(*conn);
83 }
84
85 std::lock_guard<std::mutex> lock(mutex_);
86 if (pool_.size() < config_.max_size) {
87 pool_.push_back(std::move(conn));
88 }
89 // If pool is full, connection is destroyed
90 }
91
92 // Pre-populate the pool with connections
93 void populate(ConnectionFactory factory, size_t count) {
94 std::lock_guard<std::mutex> lock(mutex_);
95 for (size_t i = 0; i < count && pool_.size() < config_.max_size; ++i) {
96 if (auto conn = factory()) {
97 pool_.push_back(std::move(conn));
98 }
99 }
100 }
101
102 // Statistics
103 size_t pool_size() const {
104 std::lock_guard<std::mutex> lock(mutex_);
105 return pool_.size();
106 }
107
108 size_t active_count() const {
109 return active_count_.load(std::memory_order_relaxed);
110 }
111

113 return total_acquired_.load(std::memory_order_relaxed);
114 }

118 }
119
120 double reuse_rate() const {
121 size_t acquired = total_acquired_.load(std::memory_order_relaxed);
122 if (acquired == 0) return 0.0;
123 return static_cast<double>(total_reused_.load(std::memory_order_relaxed)) / acquired;
124 }
125

128 std::lock_guard<std::mutex> lock(mutex_);
129 pool_.clear();
130 }
