195 return [func = std::forward<F>(f)](Request &req) mutable -> Task<Response> {

201 template <typename... Args, typename F, size_t... Is>
202 static Task<Response> invoke_with_params(F &func, Request &req,
203 std::index_sequence<Is...>) {
204 if constexpr (sizeof...(Args) == 0) {
205 // No parameters to extract
206 co_return co_await func(req);
207 } else {
208 // Extract each parameter from route_params
209 auto params = std::make_tuple(extract_param<Args>(req, Is)...);
210
211 // Check if any extraction failed
212 if (!all_valid(std::get<Is>(params)...)) {
213 co_return Response::bad_request("Invalid route parameters");
214 }
215
216 // Call the handler with extracted values
217 co_return co_await func(*std::get<Is>(params)..., req);
218 }
219 }
