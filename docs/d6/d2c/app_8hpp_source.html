63// Pre-compiled middleware chain - built once, executed many times
64class CompiledMiddlewareChain {
65 std::vector<Middleware> middleware_;
66 bool compiled_ = false;
67
68public:
69 void add(Middleware mw) {
70 middleware_.push_back(std::move(mw));
71 compiled_ = false;
72 }
73
74 bool empty() const noexcept { return middleware_.empty(); }
75 size_t size() const noexcept { return middleware_.size(); }
76
77 // Execute the chain with a final handler
78 Task<Response> execute(Request &req, Handler &handler) const {
79 if (middleware_.empty()) {
80 co_return co_await handler(req);
81 }
82
83 // Build chain from innermost (handler) to outermost (first middleware)
84 // We capture by index to avoid lambda capture issues
85 co_return co_await execute_at(0, req, handler);
86 }
87
88 // Execute with a not-found fallback
89 Task<Response> execute_or_not_found(Request &req,
90 const Handler *handler) const {
91 if (middleware_.empty()) {
92 if (handler) {
93 co_return co_await (*handler)(req);
94 }
95 co_return Response::not_found();
96 }
97
98 co_return co_await execute_at_or_not_found(0, req, handler);
99 }
100
101private:
102 Task<Response> execute_at(size_t idx, Request &req,
103 const Handler &handler) const {
104 if (idx >= middleware_.size()) {
105 co_return co_await handler(req);
106 }
107
108 // Create next function that continues the chain
109 Next next = [this, idx, &handler](Request &r) -> Task<Response> {
110 return execute_at(idx + 1, r, handler);
111 };
112
113 co_return co_await middleware_[idx](req, next);
114 }
115
116 Task<Response> execute_at_or_not_found(size_t idx, Request &req,
117 const Handler *handler) const {
118 if (idx >= middleware_.size()) {
119 if (handler) {
120 co_return co_await (*handler)(req);
121 }
122 co_return Response::not_found();
123 }
124
125 // Create next function that continues the chain
126 Next next = [this, idx, handler](Request &r) -> Task<Response> {
127 return execute_at_or_not_found(idx + 1, r, handler);
128 };
129
130 co_return co_await middleware_[idx](req, next);
131 }
132};

314 Task<Response> fetch(HttpMethod method, std::string_view route,
315 std::string body = "") {
316 // Build request
317 Request req;
318 req.set_method(method);
319 req.set_path(std::string(route));
320 req.set_body(std::move(body));
321 req.set_http_version("HTTP/1.1");
322
323 // Apply auth state (add cookies, tokens, etc.)
324 if (auth_state_) {
325 auth_state_->apply(req);
326 }
327
328#ifdef COROUTE_CLIENT_MODE
329 // Client mode: dispatch via HTTP transport to remote server
330 if (!fetch_transport_) {
331 throw std::runtime_error(
332 "fetch() called in client mode without transport configured");
333 }
334 Response resp = co_await fetch_transport_->dispatch(req);
335#else
336 // Web/server mode: dispatch in-process through middleware chain
337 auto match = router_.match(method, route);
338 if (match) {
339 req.set_route_params(std::move(match.params));
340 }
341 Response resp =
342 co_await middleware_chain_.execute_or_not_found(req, match.handler);
343#endif
344
345 // Observe response for auth state updates
346 if (auth_state_) {
347 auth_state_->observe(resp);
348 }
349
350 co_return resp;
351 }

382 req.set_body(std::move(body));
383 req.set_http_version("HTTP/1.1");
384
385 // Forward cookies from original request (web/server mode)
386 auto cookie = original.header("Cookie");
387 if (cookie) {

389 }
390
391 // Forward authorization header if present
392 auto auth = original.header("Authorization");
393 if (auth) {
394 req.add_header("Authorization", std::string(*auth));
395 }
396
397 // Apply auth state (may override with stored tokens for client mode)
398 if (auth_state_) {
399 auth_state_->apply(req);
400 }
401
402#ifdef COROUTE_CLIENT_MODE
403 if (!fetch_transport_) {
404 throw std::runtime_error(
405 "fetch() called in client mode without transport configured");
406 }
407 Response resp = co_await fetch_transport_->dispatch(req);
408#else
409 auto match = router_.match(method, route);
410 if (match) {
411 req.set_route_params(std::move(match.params));
412 }
413 Response resp =
414 co_await middleware_chain_.execute_or_not_found(req, match.handler);
415#endif
416
417 if (auth_state_) {
418 auth_state_->observe(resp);
419 }

421 co_return resp;

575 // Render a template file with data (like v1's app.render())
576 std::string render(const std::string &filename, const nlohmann::json &data) {
577 ensure_template_env();
578
579 if (template_caching_) {
580 std::lock_guard lock(template_mutex_);
581 auto it = template_cache_.find(filename);
582 if (it != template_cache_.end()) {
583 return template_env_->render(it->second, data);
584 }
585 }
586
587 auto path = template_dir_ / filename;
588 auto tmpl = template_env_->parse_template(path.string());
589 std::string result = template_env_->render(tmpl, data);
590
591 if (template_caching_) {
592 std::lock_guard lock(template_mutex_);
593 // Cache the template content
594 std::ifstream file(path);
595 if (file) {
596 std::ostringstream ss;
597 ss << file.rdbuf();
598 template_cache_[filename] = ss.str();
599 }
600 }
601
602 return result;
603 }
