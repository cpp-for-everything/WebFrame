51bool is_websocket_upgrade(const Request& req) {
52 // Check for required headers
53 auto connection = req.header("Connection");
54 auto upgrade = req.header("Upgrade");
55 auto ws_key = req.header("Sec-WebSocket-Key");
56 auto ws_version = req.header("Sec-WebSocket-Version");
57
58 if (!connection || !upgrade || !ws_key || !ws_version) {
59 return false;
60 }
61
62 // Check header values (case-insensitive for Connection and Upgrade)
63 bool has_upgrade = false;
64 std::string conn_lower(*connection);
65 for (auto& c : conn_lower) c = static_cast<char>(std::tolower(c));
66 has_upgrade = conn_lower.find("upgrade") != std::string::npos;
67
68 std::string upgrade_lower(*upgrade);
69 for (auto& c : upgrade_lower) c = static_cast<char>(std::tolower(c));
70 bool is_websocket = upgrade_lower == "websocket";
71
72 // Version must be 13
73 bool version_ok = *ws_version == "13";
74
75 return has_upgrade && is_websocket && version_ok;
76}

78std::string compute_accept_key(std::string_view client_key) {
79 // Concatenate client key with magic GUID
80 std::string combined;

82 combined.append(client_key);
83 combined.append(WS_MAGIC_GUID);

128// Frame header structure:
129// Byte 0: FIN (1 bit) + RSV1-3 (3 bits) + Opcode (4 bits)
130// Byte 1: MASK (1 bit) + Payload length (7 bits)
131// If payload length == 126: next 2 bytes are actual length (16-bit)
132// If payload length == 127: next 8 bytes are actual length (64-bit)
133// If MASK bit set: next 4 bytes are masking key
134
135struct FrameHeader {
136 bool fin = true;
137 bool rsv1 = false;
138 bool rsv2 = false;
139 bool rsv3 = false;
140 WebSocketOpcode opcode = WebSocketOpcode::Text;
141 bool masked = false;
142 uint64_t payload_length = 0;
143 std::array<uint8_t, 4> mask_key{};
144};
145
146// Parse frame header from buffer
147// Returns number of bytes consumed, or 0 if not enough data
148size_t parse_frame_header(const uint8_t* data, size_t len, FrameHeader& header) {
149 if (len < 2) return 0;
150
151 header.fin = (data[0] & 0x80) != 0;
152 header.rsv1 = (data[0] & 0x40) != 0;
153 header.rsv2 = (data[0] & 0x20) != 0;
154 header.rsv3 = (data[0] & 0x10) != 0;
155 header.opcode = static_cast<WebSocketOpcode>(data[0] & 0x0F);
156 header.masked = (data[1] & 0x80) != 0;
157
158 uint8_t len_byte = data[1] & 0x7F;
159 size_t header_size = 2;
160
161 if (len_byte <= 125) {
162 header.payload_length = len_byte;
163 } else if (len_byte == 126) {
164 if (len < 4) return 0;
165 header.payload_length = (static_cast<uint64_t>(data[2]) << 8) | data[3];
166 header_size = 4;
167 } else { // 127
168 if (len < 10) return 0;
169 header.payload_length = 0;
170 for (int i = 0; i < 8; ++i) {
171 header.payload_length = (header.payload_length << 8) | data[2 + i];
172 }
173 header_size = 10;
174 }
175
176 if (header.masked) {
177 if (len < header_size + 4) return 0;
178 std::memcpy(header.mask_key.data(), data + header_size, 4);
179 header_size += 4;
180 }
181
182 return header_size;
183}

185// Serialize frame header
186std::vector<uint8_t> serialize_frame_header(const FrameHeader& header) {
187 std::vector<uint8_t> result;
188
189 uint8_t byte0 = static_cast<uint8_t>(header.opcode);
190 if (header.fin) byte0 |= 0x80;
191 if (header.rsv1) byte0 |= 0x40;
192 if (header.rsv2) byte0 |= 0x20;
193 if (header.rsv3) byte0 |= 0x10;
194 result.push_back(byte0);
195
196 uint8_t byte1 = header.masked ? 0x80 : 0x00;
197 if (header.payload_length <= 125) {
198 byte1 |= static_cast<uint8_t>(header.payload_length);
199 result.push_back(byte1);
200 } else if (header.payload_length <= 0xFFFF) {
201 byte1 |= 126;
202 result.push_back(byte1);
203 result.push_back(static_cast<uint8_t>(header.payload_length >> 8));
204 result.push_back(static_cast<uint8_t>(header.payload_length & 0xFF));
205 } else {
206 byte1 |= 127;
207 result.push_back(byte1);
208 for (int i = 7; i >= 0; --i) {
209 result.push_back(static_cast<uint8_t>(header.payload_length >> (i * 8)));
210 }
211 }
212
213 if (header.masked) {
214 result.insert(result.end(), header.mask_key.begin(), header.mask_key.end());
215 }
216
217 return result;
218}

220// Apply/remove XOR mask to payload
221void apply_mask(uint8_t* data, size_t len, const std::array<uint8_t, 4>& mask) {
222 for (size_t i = 0; i < len; ++i) {
223 data[i] ^= mask[i % 4];
224 }
225}

248 Task<expected<WebSocketMessage, Error>> receive() override {
249 while (is_open_) {
250 // Try to parse a frame from existing buffer
251 if (!read_buffer_.empty()) {
252 FrameHeader header;
253 size_t header_size = parse_frame_header(read_buffer_.data(), read_buffer_.size(), header);
254
255 if (header_size > 0 && read_buffer_.size() >= header_size + header.payload_length) {
256 // We have a complete frame
257 std::vector<uint8_t> payload(
258 read_buffer_.begin() + header_size,
259 read_buffer_.begin() + header_size + header.payload_length
260 );
261
262 // Remove frame from buffer
263 read_buffer_.erase(
264 read_buffer_.begin(),
265 read_buffer_.begin() + header_size + header.payload_length
266 );
267
268 // Unmask if needed (client frames must be masked)
269 if (header.masked) {
270 apply_mask(payload.data(), payload.size(), header.mask_key);
271 }
272
273 // Handle control frames immediately
274 if (header.opcode == WebSocketOpcode::Ping) {
275 co_await pong(payload);
276 continue;
277 }
278
279 if (header.opcode == WebSocketOpcode::Pong) {
280 // Ignore pongs for now
281 continue;
282 }
283
284 if (header.opcode == WebSocketOpcode::Close) {
285 is_open_ = false;
286 // Echo close frame
287 WebSocketCloseCode code = WebSocketCloseCode::Normal;
288 std::string_view reason;
289 if (payload.size() >= 2) {
290 code = static_cast<WebSocketCloseCode>((payload[0] << 8) | payload[1]);
291 if (payload.size() > 2) {
292 reason = std::string_view(
293 reinterpret_cast<const char*>(payload.data() + 2),
294 payload.size() - 2
295 );
296 }
297 }
298 co_await send_close_frame(code, reason);
299 co_return WebSocketMessage{WebSocketOpcode::Close, std::move(payload)};
300 }
301
302 // Handle data frames
303 if (header.opcode == WebSocketOpcode::Continuation) {
304 // Continuation of fragmented message
305 fragment_buffer_.insert(fragment_buffer_.end(), payload.begin(), payload.end());
306
307 if (fragment_buffer_.size() > max_message_size_) {
308 co_await close(WebSocketCloseCode::MessageTooBig, "Message too large");
309 co_return unexpected(Error::io(IoError::Unknown, "Message too large"));
310 }
311
312 if (header.fin) {
313 // Final fragment
314 WebSocketMessage msg{fragment_opcode_, std::move(fragment_buffer_)};
315 fragment_buffer_.clear();
316 co_return msg;
317 }
318 } else {
319 // New message
320 if (!fragment_buffer_.empty()) {
321 // Protocol error: new message while fragmented message incomplete
322 co_await close(WebSocketCloseCode::ProtocolError, "Incomplete fragmented message");
323 co_return unexpected(Error::io(IoError::Unknown, "Protocol error"));
324 }
325
326 if (header.fin) {
327 // Complete message in single frame
328 co_return WebSocketMessage{header.opcode, std::move(payload)};
329 } else {
330 // Start of fragmented message
331 fragment_opcode_ = header.opcode;
332 fragment_buffer_ = std::move(payload);
333 }
334 }
335
336 continue;
337 }
338 }
339
340 // Need more data
341 std::array<uint8_t, 8192> buf;
342 auto result = co_await conn_->async_read(buf.data(), buf.size());
343 if (!result) {
344 is_open_ = false;
345 co_return unexpected(result.error());
346 }
347 if (*result == 0) {
348 is_open_ = false;
349 co_return unexpected(Error::io(IoError::ConnectionReset, "Connection closed"));
350 }
351
352 read_buffer_.insert(read_buffer_.end(), buf.begin(), buf.begin() + *result);
353 }
354
355 co_return unexpected(Error::io(IoError::ConnectionReset, "Connection closed"));
356 }

390 Task<expected<void, Error>> send_frame(WebSocketOpcode opcode, const uint8_t* data, size_t len) {
391 FrameHeader header;
392 header.fin = true;
393 header.opcode = opcode;
394 header.masked = false; // Server frames are not masked
395 header.payload_length = len;
396
397 auto header_bytes = serialize_frame_header(header);
398
399 // Send header
400 auto header_result = co_await conn_->async_write_all(header_bytes.data(), header_bytes.size());
401 if (!header_result) co_return unexpected(header_result.error());
402
403 // Send payload
404 if (len > 0) {
405 auto payload_result = co_await conn_->async_write_all(data, len);
406 if (!payload_result) co_return unexpected(payload_result.error());
407 }
408
409 co_return expected<void, Error>{};
410 }

412 Task<expected<void, Error>> send_close_frame(WebSocketCloseCode code, std::string_view reason) {
413 std::vector<uint8_t> payload;
414 payload.push_back(static_cast<uint8_t>(static_cast<uint16_t>(code) >> 8));
415 payload.push_back(static_cast<uint8_t>(static_cast<uint16_t>(code) & 0xFF));
416 payload.insert(payload.end(), reason.begin(), reason.end());
417
418 co_return co_await send_frame(WebSocketOpcode::Close, payload.data(), payload.size());

427upgrade_to_websocket(std::unique_ptr<Connection> conn, const Request& req) {
428 if (!is_websocket_upgrade(req)) {
429 co_return unexpected(Error::io(IoError::InvalidArgument, "Not a WebSocket upgrade request"));
430 }
431
432 // Create and send upgrade response
433 auto response = create_upgrade_response(req);
434
435 // Serialize response
436 std::string resp_str;
437 resp_str += "HTTP/1.1 ";
438 resp_str += std::to_string(response.status());
439 resp_str += " ";
440 resp_str += response.status_text();
441 resp_str += "\r\n";
442
443 for (const auto& [name, value] : response.headers()) {
444 resp_str += name;
445 resp_str += ": ";
446 resp_str += value;
447 resp_str += "\r\n";
448 }
449 resp_str += "\r\n";
450
451 auto result = co_await conn->async_write_all(resp_str.data(), resp_str.size());
452 if (!result) {
453 co_return unexpected(result.error());
454 }
455
456 co_return std::make_unique<WebSocketConnectionImpl>(std::move(conn));
457}
