142// Send file using zero-copy (TransmitFile/sendfile)
143// Returns bytes sent or error
144inline Task<net::TransmitResult> send_file_zero_copy(
145 net::Connection& conn,

163// Send response headers then file body using zero-copy
164// Useful for static file serving
165inline Task<expected<size_t, Error>> send_response_with_file(
166 net::Connection& conn,
167 Response& headers_response,
168 const std::filesystem::path& file_path,
169 size_t offset = 0,
170 size_t length = 0)
171{
172 // Send headers first
173 std::string headers = headers_response.serialize();
174 auto write_result = co_await conn.async_write_all(headers.data(), headers.size());
175 if (!write_result) {
176 co_return unexpected(write_result.error());
177 }
178
179 // Then send file body using zero-copy
180 auto transmit_result = co_await send_file_zero_copy(conn, file_path, offset, length);
181 if (!transmit_result) {
182 co_return unexpected(transmit_result.error());
183 }
184
185 co_return *write_result + *transmit_result;
186}
