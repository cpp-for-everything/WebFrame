82 // Add stored cookies to request
83 if (!cookies_.empty()) {
84 std::string cookie_header;
85 for (const auto &[name, value] : cookies_) {
86 if (!cookie_header.empty()) {

151 void parse_set_cookie(const std::string &header) {
152 // Simple parsing: extract name=value before first semicolon
153 auto semi_pos = header.find(';');
154 std::string name_value =
155 (semi_pos != std::string::npos) ? header.substr(0, semi_pos) : header;
156
157 auto eq_pos = name_value.find('=');
158 if (eq_pos != std::string::npos) {
159 std::string name = name_value.substr(0, eq_pos);
160 std::string value = name_value.substr(eq_pos + 1);
161
162 // Trim whitespace
163 while (!name.empty() && name.front() == ' ')
164 name.erase(0, 1);
165 while (!name.empty() && name.back() == ' ')
166 name.pop_back();
167 while (!value.empty() && value.front() == ' ')
168 value.erase(0, 1);
169 while (!value.empty() && value.back() == ' ')
170 value.pop_back();
171
172 if (!name.empty()) {
173 cookies_[name] = value;
174 }
175 }
176 }
