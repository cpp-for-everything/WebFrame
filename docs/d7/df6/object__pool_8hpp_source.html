16class ObjectPool {
17public:
18 using ResetFunc = std::function<void(T&)>;
19
20private:

25
26public:
27 explicit ObjectPool(size_t max_size = 1024, ResetFunc reset_func = nullptr)
28 : max_size_(max_size)
29 , reset_func_(std::move(reset_func))
30 {
31 pool_.reserve(std::min(max_size, size_t(64)));
32 }
33
34 // Pre-allocate objects
35 void reserve(size_t count) {
36 std::lock_guard<std::mutex> lock(mutex_);
37 while (pool_.size() < count && pool_.size() < max_size_) {
38 pool_.push_back(std::make_unique<T>());
39 }
40 }
41
42 // Acquire an object from the pool (or create new)
43 std::unique_ptr<T> acquire() {
44 std::lock_guard<std::mutex> lock(mutex_);
45 if (pool_.empty()) {
46 return std::make_unique<T>();
47 }
48 auto obj = std::move(pool_.back());
49 pool_.pop_back();
50 return obj;
51 }
52
53 // Release an object back to the pool
54 void release(std::unique_ptr<T> obj) {
55 if (!obj) return;
56
57 // Reset object state
58 if (reset_func_) {
59 reset_func_(*obj);
60 }
61
62 std::lock_guard<std::mutex> lock(mutex_);
63 if (pool_.size() < max_size_) {
64 pool_.push_back(std::move(obj));
65 }
66 // If pool is full, object is destroyed
67 }
68
69 // Get current pool size
70 size_t size() const {
71 std::lock_guard<std::mutex> lock(mutex_);
72 return pool_.size();
73 }
74
75 // Get max pool size
76 size_t max_size() const { return max_size_; }
77
78 // Clear the pool
79 void clear() {
80 std::lock_guard<std::mutex> lock(mutex_);
81 pool_.clear();
82 }
83};
