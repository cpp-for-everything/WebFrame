28 return cancelled_.load(std::memory_order_acquire);

30
31 bool cancel() noexcept {
32 bool expected = false;
33 if (cancelled_.compare_exchange_strong(expected, true, std::memory_order_acq_rel)) {
34 // Successfully cancelled, invoke callbacks
35 std::vector<std::function<void()>> cbs;
36 {
37 std::lock_guard lock(mutex_);
38 cbs = std::move(callbacks_);
39 }
40 for (auto& cb : cbs) {
41 if (cb) cb();
42 }
43 return true;
44 }
45 return false; // Already cancelled
46 }

49 // Returns true if registered, false if already cancelled (callback invoked immediately)

54 }
55

57 if (cancelled_.load(std::memory_order_acquire)) {
58 if (cb) cb();
59 return false;
60 }
61 callbacks_.push_back(std::move(cb));
62 return true;
63 }
