34expected<FrameHeader, Error> FrameHeader::parse(std::span<const uint8_t> data) {
35 if (data.size() < Constants::FrameHeaderSize) {
36 return unexpected(Error::io(IoError::InvalidArgument, "Frame header too short"));
37 }
38
39 FrameHeader header;
40
41 // Length (24 bits, big-endian)
42 header.length = (static_cast<uint32_t>(data[0]) << 16) |
43 (static_cast<uint32_t>(data[1]) << 8) |
44 static_cast<uint32_t>(data[2]);
45
46 // Type (8 bits)
47 header.type = static_cast<FrameType>(data[3]);

54 (static_cast<uint32_t>(data[6]) << 16) |
55 (static_cast<uint32_t>(data[7]) << 8) |

65std::vector<uint8_t> serialize_settings_frame(
66 std::span<const SettingsEntry> settings,
67 bool ack
68) {
69 // Each setting is 6 bytes (2 byte ID + 4 byte value)
70 size_t payload_size = ack ? 0 : settings.size() * 6;
71
72 FrameHeader header;
73 header.length = static_cast<uint32_t>(payload_size);
74 header.type = FrameType::Settings;
75 header.flags = ack ? FrameFlags::Ack : 0;
76 header.stream_id = 0; // SETTINGS must be on stream 0
77
78 std::vector<uint8_t> result;
79 result.reserve(Constants::FrameHeaderSize + payload_size);
80
81 // Add header
82 auto header_bytes = header.serialize();
83 result.insert(result.end(), header_bytes.begin(), header_bytes.end());
84
85 // Add settings (if not ACK)
86 if (!ack) {
87 for (const auto& setting : settings) {
88 // ID (16 bits, big-endian)
89 result.push_back(static_cast<uint8_t>(static_cast<uint16_t>(setting.id) >> 8));
90 result.push_back(static_cast<uint8_t>(static_cast<uint16_t>(setting.id) & 0xFF));
91
92 // Value (32 bits, big-endian)
93 result.push_back(static_cast<uint8_t>(setting.value >> 24));
94 result.push_back(static_cast<uint8_t>((setting.value >> 16) & 0xFF));
95 result.push_back(static_cast<uint8_t>((setting.value >> 8) & 0xFF));
96 result.push_back(static_cast<uint8_t>(setting.value & 0xFF));
97 }
98 }
99
100 return result;
101}
