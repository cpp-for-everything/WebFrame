49expected<Request, Error> Stream::build_request() const {
50 if (!headers_complete_) {
51 return unexpected(Error::io(IoError::InvalidArgument, "Headers not complete"));
52 }
53
54 // Extract method
55 auto method_str = get_method(request_headers_);
56 if (method_str.empty()) {
57 return unexpected(Error::io(IoError::InvalidArgument, "Missing :method"));
58 }
59
60 HttpMethod method;
61 if (method_str == "GET") method = HttpMethod::GET;
62 else if (method_str == "POST") method = HttpMethod::POST;
63 else if (method_str == "PUT") method = HttpMethod::PUT;
64 else if (method_str == "DELETE") method = HttpMethod::DELETE;
65 else if (method_str == "HEAD") method = HttpMethod::HEAD;
66 else if (method_str == "OPTIONS") method = HttpMethod::OPTIONS;
67 else if (method_str == "PATCH") method = HttpMethod::PATCH;
68 else {
69 return unexpected(Error::io(IoError::InvalidArgument, "Unknown method"));
70 }
71
72 // Extract path
73 auto path = get_path(request_headers_);
74 if (path.empty()) {
75 return unexpected(Error::io(IoError::InvalidArgument, "Missing :path"));
76 }
77
78 // Build request
79 Request req;
80 req.set_method(method);
81
82 // Parse path and query string
83 auto query_pos = path.find('?');
84 if (query_pos != std::string_view::npos) {
85 req.set_path(std::string(path.substr(0, query_pos)));
86 req.set_query_string(std::string(path.substr(query_pos + 1)));
87 } else {
88 req.set_path(std::string(path));
89 }
90
91 // Copy headers (excluding pseudo-headers)
92 for (const auto& h : request_headers_) {
93 if (!h.is_pseudo()) {
94 req.add_header(h.name, h.value);
95 }
96 }
97
98 // Set authority as Host header if present
99 auto authority = get_authority(request_headers_);
100 if (!authority.empty()) {
101 req.add_header("Host", std::string(authority));
102 }
103
104 // Set body
105 if (!request_body_.empty()) {
106 req.set_body(std::string(
107 reinterpret_cast<const char*>(request_body_.data()),
108 request_body_.size()
109 ));
110 }
111
112 // Mark as HTTP/2
113 req.set_http_version("HTTP/2");
114
115 return req;
116}

118Task<expected<void, Error>> Stream::send_response(const Response& response) {
119 if (response_headers_sent_) {
120 co_return unexpected(Error::io(IoError::InvalidArgument, "Response already sent"));
121 }
122
123 // Build response headers
124 std::vector<Header> headers;
125
126 // :status pseudo-header
127 headers.push_back({":status", std::to_string(response.status())});
128
129 // Regular headers
130 for (const auto& [name, value] : response.headers()) {
131 // Convert header name to lowercase
132 std::string lower_name;
133 lower_name.reserve(name.size());
134 for (char c : name) {
135 lower_name.push_back(static_cast<char>(std::tolower(static_cast<unsigned char>(c))));
136 }
137
138 // Skip connection-specific headers (not allowed in HTTP/2)
139 if (lower_name == "connection" || lower_name == "keep-alive" ||
140 lower_name == "transfer-encoding" || lower_name == "upgrade") {
141 continue;
142 }
143
144 headers.push_back({std::move(lower_name), value});
145 }
146
147 // Add content-length if body present and not already set
148 const auto& body = response.body();
149 bool has_content_length = false;
150 for (const auto& h : headers) {
151 if (h.name == "content-length") {
152 has_content_length = true;
153 break;
154 }
155 }
156 if (!has_content_length && !body.empty()) {
157 headers.push_back({"content-length", std::to_string(body.size())});
158 }
159
160 // Send headers
161 bool end_stream = body.empty();
162 auto result = co_await send_headers(headers, end_stream);
163 if (!result) {
164 co_return unexpected(result.error());
165 }
166
167 response_headers_sent_ = true;
168
169 // Send body if present
170 if (!body.empty()) {
171 auto body_result = co_await send_data(
172 std::span<const uint8_t>(
173 reinterpret_cast<const uint8_t*>(body.data()),
174 body.size()
175 ),
176 true // end_stream
177 );
178 if (!body_result) {
179 co_return unexpected(body_result.error());
180 }
181 }
182
183 response_complete_ = true;
184 transition_to(StreamState::Closed);
185
186 co_return expected<void, Error>{};
187}

218Task<expected<void, Error>> Stream::send_data(
219 std::span<const uint8_t> data,
220 bool end_stream
221) {
222 const uint32_t max_frame_size = connection_->remote_settings().max_frame_size;
223
224 size_t offset = 0;
225 while (offset < data.size()) {
226 // Calculate chunk size (respecting frame size and flow control)
227 size_t remaining = data.size() - offset;
228 size_t chunk_size = std::min(remaining, static_cast<size_t>(max_frame_size));
229
230 // TODO: Check flow control window and wait if necessary
231 // For now, assume we have enough window
232
233 bool is_last = (offset + chunk_size >= data.size());
234 bool frame_end_stream = end_stream && is_last;
235
236 auto frame = serialize_data_frame(
237 id_,
238 data.subspan(offset, chunk_size),
239 frame_end_stream
240 );
241
242 auto result = co_await connection_->send_frame(frame);
243 if (!result) {
244 co_return unexpected(result.error());
245 }
246
247 offset += chunk_size;
248 }
