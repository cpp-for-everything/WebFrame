44int64_t parse_chunk_size(std::string_view hex_size) {
45 // Trim whitespace
46 while (!hex_size.empty() && std::isspace(static_cast<unsigned char>(hex_size.front()))) {
47 hex_size.remove_prefix(1);
48 }
49 while (!hex_size.empty() && std::isspace(static_cast<unsigned char>(hex_size.back()))) {
50 hex_size.remove_suffix(1);
51 }
52
53 // Handle chunk extensions (ignore everything after semicolon)
54 auto semicolon = hex_size.find(';');
55 if (semicolon != std::string_view::npos) {
56 hex_size = hex_size.substr(0, semicolon);
57 }
58
59 if (hex_size.empty()) {
60 return -1;
61 }
62
63 int64_t result = 0;
64 for (char c : hex_size) {
65 result *= 16;
66 if (c >= '0' && c <= '9') {
67 result += c - '0';
68 } else if (c >= 'a' && c <= 'f') {
69 result += c - 'a' + 10;
70 } else if (c >= 'A' && c <= 'F') {
71 result += c - 'A' + 10;
72 } else {
73 return -1; // Invalid character
74 }
75 }
76
77 return result;
78}

98Task<expected<void, Error>> ChunkedResponse::send_headers() {
99 if (!conn_) {
100 co_return unexpected(Error::http(HttpError::Internal, "No connection set"));
101 }
102
103 if (headers_sent_) {
104 co_return expected<void, Error>{}; // Already sent
105 }
106
107 std::ostringstream oss;
108
109 // Status line
110 oss << "HTTP/1.1 " << status_ << " " << status_text(status_) << "\r\n";
111
112 // Add Transfer-Encoding header
113 oss << "Transfer-Encoding: chunked\r\n";
114
115 // Add Trailer header if we have trailers
116 if (!trailers_.empty()) {
117 oss << "Trailer: ";
118 for (size_t i = 0; i < trailers_.size(); ++i) {
119 if (i > 0) oss << ", ";
120 oss << trailers_[i].first;
121 }
122 oss << "\r\n";
123 }
124
125 // User headers
126 for (const auto& [key, value] : headers_) {
127 // Skip Content-Length (incompatible with chunked)
128 if (key == "Content-Length") continue;
129 // Skip Transfer-Encoding (we set it)
130 if (key == "Transfer-Encoding") continue;
131 oss << key << ": " << value << "\r\n";
132 }
133
134 // End of headers
135 oss << "\r\n";
136
137 std::string header_data = oss.str();
138 auto result = co_await conn_->async_write_all(header_data.data(), header_data.size());
139
140 if (!result) {
141 co_return unexpected(result.error());
142 }
143
144 headers_sent_ = true;
145 co_return expected<void, Error>{};
146}

220Task<expected<std::string, Error>> ChunkedBodyReader::read_line() {
221 std::string line;
222
223 // First check buffer for existing data
224 auto crlf_pos = buffer_.find("\r\n");
225 if (crlf_pos != std::string::npos) {
226 line = buffer_.substr(0, crlf_pos);
227 buffer_.erase(0, crlf_pos + 2);
228 co_return line;
229 }
230
231 // Read more data
232 char buf[1024];
233 while (true) {
234 auto result = co_await conn_->async_read(buf, sizeof(buf));
235 if (!result) {
236 co_return unexpected(result.error());
237 }
238
239 if (*result == 0) {
240 // Connection closed
241 co_return unexpected(Error::http(HttpError::BadRequest, "Connection closed while reading chunk"));
242 }
243
244 buffer_.append(buf, *result);
245
246 crlf_pos = buffer_.find("\r\n");
247 if (crlf_pos != std::string::npos) {
248 line = buffer_.substr(0, crlf_pos);
249 buffer_.erase(0, crlf_pos + 2);
250 co_return line;
251 }
252
253 // Prevent buffer from growing too large
254 if (buffer_.size() > 8192) {
255 co_return unexpected(Error::http(HttpError::BadRequest, "Chunk size line too long"));
256 }

260Task<expected<std::string, Error>> ChunkedBodyReader::read_bytes(size_t n) {
261 std::string result;
262 result.reserve(n);
263
264 // Use buffered data first
265 if (!buffer_.empty()) {
266 size_t to_use = std::min(buffer_.size(), n);
267 result.append(buffer_, 0, to_use);
268 buffer_.erase(0, to_use);
269 }
270
271 // Read remaining
272 while (result.size() < n) {
273 char buf[4096];
274 size_t to_read = std::min(sizeof(buf), n - result.size());
275

277 if (!read_result) {
278 co_return unexpected(read_result.error());
279 }
280
281 if (*read_result == 0) {
282 co_return unexpected(Error::http(HttpError::BadRequest, "Connection closed while reading chunk data"));
283 }
284
285 result.append(buf, *read_result);
286 }

288 co_return result;

291Task<expected<std::string, Error>> ChunkedBodyReader::read_chunk() {
292 if (finished_) {
293 co_return std::string{};
294 }
295
296 // Read chunk size line
297 auto size_line = co_await read_line();
298 if (!size_line) {
299 co_return unexpected(size_line.error());
300 }
301
302 int64_t chunk_size = chunked::parse_chunk_size(*size_line);
303 if (chunk_size < 0) {
304 co_return unexpected(Error::http(HttpError::BadRequest, "Invalid chunk size"));
305 }
306
307 // Final chunk
308 if (chunk_size == 0) {
309 // Read trailers (if any) until empty line
310 while (true) {
311 auto trailer_line = co_await read_line();
312 if (!trailer_line) {
313 co_return unexpected(trailer_line.error());
314 }
315
316 if (trailer_line->empty()) {
317 break; // End of trailers
318 }
319
320 // Parse trailer
321 auto colon = trailer_line->find(':');
322 if (colon != std::string::npos) {
323 std::string key = trailer_line->substr(0, colon);
324 std::string value = trailer_line->substr(colon + 1);
325 // Trim leading whitespace from value
326 while (!value.empty() && std::isspace(static_cast<unsigned char>(value.front()))) {
327 value.erase(0, 1);
328 }
329 trailers_.emplace_back(std::move(key), std::move(value));
330 }
331 }
332
333 finished_ = true;
334 co_return std::string{};
335 }
336
337 // Read chunk data
338 auto chunk_data = co_await read_bytes(static_cast<size_t>(chunk_size));
339 if (!chunk_data) {
340 co_return unexpected(chunk_data.error());
341 }
342
343 // Read trailing CRLF
344 auto crlf = co_await read_bytes(2);
345 if (!crlf) {
346 co_return unexpected(crlf.error());
347 }
348
349 if (*crlf != "\r\n") {
350 co_return unexpected(Error::http(HttpError::BadRequest, "Missing CRLF after chunk data"));
351 }
352
353 co_return std::move(*chunk_data);
354}

356Task<expected<std::string, Error>> ChunkedBodyReader::read_all(size_t max_size) {
357 std::string result;
358
359 while (!finished_) {
360 auto chunk = co_await read_chunk();
361 if (!chunk) {
362 co_return unexpected(chunk.error());
363 }
364
365 if (chunk->empty()) {
366 break; // Final chunk
367 }
368
369 if (result.size() + chunk->size() > max_size) {
370 co_return unexpected(Error::http(HttpError::PayloadTooLarge, "Chunked body too large"));
371 }
372
373 result.append(*chunk);
374 }
375
376 co_return result;
377}

387{
388 ChunkedResponse resp(conn);

390
