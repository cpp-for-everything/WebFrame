12Histogram::Histogram(std::string name, std::vector<double> buckets, std::string help)
13 : name_(std::move(name))
14 , help_(std::move(help))
15 , buckets_(std::move(buckets))
16{
17 // Sort buckets
18 std::sort(buckets_.begin(), buckets_.end());
19
20 // Add +Inf bucket
21 buckets_.push_back(std::numeric_limits<double>::infinity());
22
23 // Initialize bucket counts
24 bucket_count_size_ = buckets_.size();
25 bucket_counts_ = std::make_unique<std::atomic<uint64_t>[]>(bucket_count_size_);
26 for (size_t i = 0; i < bucket_count_size_; ++i) {
27 bucket_counts_[i].store(0, std::memory_order_relaxed);
28 }

31void Histogram::observe(double value) {
32 // Increment count and sum
33 count_.fetch_add(1, std::memory_order_relaxed);
34
35 double old_sum = sum_.load(std::memory_order_relaxed);
36 while (!sum_.compare_exchange_weak(old_sum, old_sum + value, std::memory_order_relaxed));
37
38 // Find bucket and increment
39 for (size_t i = 0; i < buckets_.size(); ++i) {
40 if (value <= buckets_[i]) {
41 bucket_counts_[i].fetch_add(1, std::memory_order_relaxed);
42 break;
43 }
44 }
45}

95std::string MetricsRegistry::prometheus_export() const {
96 std::lock_guard<std::mutex> lock(mutex_);
97 std::ostringstream out;
98
99 // Export counters
100 for (const auto& [name, counter] : counters_) {
101 if (!counter->help().empty()) {
102 out << "# HELP " << name << " " << counter->help() << "\n";

104 out << "# TYPE " << name << " counter\n";
105 out << name;
106 if (!counter->labels().empty()) {

108 bool first = true;

112 first = false;
113 }
114 out << "}";
115 }
116 out << " " << counter->value() << "\n";
117 }
118
119 // Export gauges
120 for (const auto& [name, gauge] : gauges_) {
121 if (!gauge->help().empty()) {
122 out << "# HELP " << name << " " << gauge->help() << "\n";
123 }
124 out << "# TYPE " << name << " gauge\n";
125 out << name << " " << std::fixed << std::setprecision(6) << gauge->value() << "\n";
126 }
127
128 // Export histograms
129 for (const auto& [name, hist] : histograms_) {
130 if (!hist->help().empty()) {
131 out << "# HELP " << name << " " << hist->help() << "\n";
132 }
133 out << "# TYPE " << name << " histogram\n";
134
135 uint64_t cumulative = 0;
136 const auto& buckets = hist->buckets();
137 for (size_t i = 0; i < buckets.size(); ++i) {
138 cumulative += hist->bucket_count(i);
139 out << name << "_bucket{le=\"";
140 if (std::isinf(buckets[i])) {
141 out << "+Inf";
142 } else {
143 out << std::fixed << std::setprecision(3) << buckets[i];
144 }
145 out << "\"} " << cumulative << "\n";
146 }
147 out << name << "_sum " << std::fixed << std::setprecision(6) << hist->sum() << "\n";
148 out << name << "_count " << hist->count() << "\n";
149 }

193Middleware metrics_middleware(MetricsRegistry& registry, MetricsOptions options) {
194 auto http_metrics = std::make_shared<HttpMetrics>(registry);
195 auto registry_ptr = &registry;
196
197 return [http_metrics, registry_ptr, options](Request& req, Next next) -> Task<Response> {
198 // Check if this is a metrics endpoint request
199 if (options.expose_endpoint && req.path() == options.path) {
200 std::string body = registry_ptr->prometheus_export();
201 Response resp = Response::ok(std::move(body), "text/plain; version=0.0.4");
202 co_return resp;
203 }
204
205 // Track request
206 http_metrics->requests_total->inc();
207 http_metrics->requests_in_flight->inc();
208
209 auto start = std::chrono::steady_clock::now();
210
211 // Call next handler
212 Response resp = co_await next(req);
213
214 // Record duration
215 auto end = std::chrono::steady_clock::now();
216 auto duration = std::chrono::duration<double>(end - start).count();
217 http_metrics->request_duration->observe(duration);
218
219 // Track by status (simplified - would need labeled counter for proper impl)
220 http_metrics->requests_in_flight->dec();
221
222 co_return resp;
223 };
