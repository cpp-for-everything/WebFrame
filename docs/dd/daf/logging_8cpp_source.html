53std::string format_timestamp(std::chrono::system_clock::time_point tp) {
54 auto time_t_val = std::chrono::system_clock::to_time_t(tp);

56 tp.time_since_epoch()) % 1000;

66void ConsoleSink::write(const LogEntry& entry) {
67 std::lock_guard<std::mutex> lock(mutex_);
68
69 std::ostringstream oss;

71 // Timestamp

215Middleware request_logger(Logger& logger, RequestLogOptions options) {
216 return [&logger, options](Request& req, Next next) -> Task<Response> {
217 auto start = std::chrono::steady_clock::now();
218
219 // Call next handler
220 Response resp = co_await next(req);
221
222 auto end = std::chrono::steady_clock::now();
223 auto duration_ms = std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count();
224
225 // Build log entry
226 auto entry = logger.entry(options.level, "");
227
228 // Format based on style
229 std::ostringstream msg;
230
231 if (options.format == "dev") {
232 msg << method_to_string(req.method()) << " " << req.path()
233 << " " << resp.status() << " " << duration_ms << "ms";
234 } else if (options.format == "short") {
235 msg << req.path() << " " << resp.status() << " " << duration_ms << "ms";
236 } else {
237 // combined/common format
238 msg << method_to_string(req.method()) << " " << req.path();
239 if (!req.query_string().empty()) {
240 msg << "?" << req.query_string();
241 }
242 msg << " " << req.http_version() << " " << resp.status();
243 }
244
245 entry.message = msg.str();
246 entry.field("method", std::string(method_to_string(req.method())));
247 entry.field("path", std::string(req.path()));
248 entry.field("status", resp.status());
249 entry.field("duration_ms", duration_ms);
250
251 if (options.log_headers) {
252 for (const auto& [key, value] : req.headers()) {
253 entry.field("req_" + key, value);
254 }
255 }
256
257 logger.log(entry);
258
259 co_return resp;
260 };
