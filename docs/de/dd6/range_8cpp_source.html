16bool ByteRange::normalize(int64_t total_size) {
17 if (total_size <= 0) {
18 return false;
19 }
20
21 // Suffix range: "-500" means last 500 bytes
22 if (!start.has_value() && end.has_value()) {
23 int64_t suffix_length = end.value();
24 if (suffix_length <= 0) {
25 return false;
26 }
27 if (suffix_length >= total_size) {
28 // Entire file
29 start = 0;
30 end = total_size - 1;
31 } else {
32 start = total_size - suffix_length;
33 end = total_size - 1;
34 }
35 return true;
36 }
37
38 // Must have start at this point
39 if (!start.has_value()) {
40 return false;
41 }
42
43 int64_t s = start.value();
44
45 // Start beyond file size
46 if (s >= total_size) {
47 return false;
48 }
49
50 // Negative start is invalid
51 if (s < 0) {
52 return false;
53 }
54
55 // Open-ended range: "500-" means from 500 to end
56 if (!end.has_value()) {
57 end = total_size - 1;
58 return true;
59 }
60
61 int64_t e = end.value();
62
63 // End before start
64 if (e < s) {
65 return false;
66 }
67
68 // Clamp end to file size
69 if (e >= total_size) {
70 end = total_size - 1;
71 }
72
73 return true;
74}

99// Parse a single range spec like "0-499", "500-", "-500"
100std::optional<ByteRange> parse_range_spec(std::string_view spec) {
101 spec = trim(spec);
102 if (spec.empty()) {
103 return std::nullopt;
104 }
105
106 auto dash = spec.find('-');
107 if (dash == std::string_view::npos) {
108 return std::nullopt;
109 }
110
111 ByteRange range;
112
113 // Parse start (before dash)
114 auto start_str = trim(spec.substr(0, dash));
115 if (!start_str.empty()) {
116 try {
117 range.start = std::stoll(std::string(start_str));
118 } catch (...) {
119 return std::nullopt;
120 }
121 }
122
123 // Parse end (after dash)
124 auto end_str = trim(spec.substr(dash + 1));
125 if (!end_str.empty()) {
126 try {
127 range.end = std::stoll(std::string(end_str));
128 } catch (...) {
129 return std::nullopt;
130 }
131 }
132
133 // Must have at least start or end
134 if (!range.start.has_value() && !range.end.has_value()) {
135 return std::nullopt;
136 }
137
138 return range;
139}

145std::optional<RangeHeader> parse(std::string_view header) {
146 header = trim(header);
147 if (header.empty()) {
148 return std::nullopt;
149 }
150
151 // Find "bytes="
152 auto eq = header.find('=');
153 if (eq == std::string_view::npos) {
154 return std::nullopt;
155 }
156
157 RangeHeader result;
158 result.unit = std::string(trim(header.substr(0, eq)));
159
160 // Only support bytes
161 if (result.unit != "bytes") {
162 return std::nullopt;
163 }
164
165 // Parse range specs (comma-separated)
166 auto specs = header.substr(eq + 1);
167 size_t start = 0;
168
169 while (start < specs.size()) {
170 auto comma = specs.find(',', start);
171 auto spec = (comma == std::string_view::npos)
172 ? specs.substr(start)
173 : specs.substr(start, comma - start);
174
175 auto range = parse_range_spec(spec);
176 if (range) {
177 result.ranges.push_back(*range);
178 }
179
180 if (comma == std::string_view::npos) {
181 break;
182 }
183 start = comma + 1;
184 }
185
186 if (result.ranges.empty()) {
187 return std::nullopt;
188 }
189
190 return result;
191}

205bool check_if_range(const Request& req,
206 std::string_view etag,
207 std::string_view last_modified) {
208 auto if_range = req.header("If-Range");
209 if (!if_range) {
210 // No If-Range header means always serve range
211 return true;
212 }
213
214 std::string_view value = *if_range;
215
216 // If it looks like an ETag (starts with " or W/)
217 if (value.starts_with('"') || value.starts_with("W/")) {
218 // Compare ETags
219 return !etag.empty() && value == etag;
220 }
221
222 // Otherwise treat as date - compare with Last-Modified
223 // For simplicity, just do string comparison
224 // A proper implementation would parse HTTP dates
225 return !last_modified.empty() && value == last_modified;
226}

400Response RangeResponseBuilder::build(const Request& req) {
401 // Get total size
402 int64_t total_size;
403 if (use_file_) {
404 std::ifstream file(file_path_, std::ios::binary | std::ios::ate);
405 if (!file) {
406 return Response::not_found();
407 }
408 total_size = file.tellg();
409 } else {
410 total_size = static_cast<int64_t>(content_.size());
411 }
412
413 // Check for Range header
414 auto range_header = range::get_range(req);
415 if (!range_header || !range_header->is_valid()) {
416 return build_full();
417 }
418
419 // Check If-Range
420 if (!range::check_if_range(req, etag_, last_modified_)) {
421 return build_full();
422 }
423
424 // Normalize and validate ranges
425 std::vector<ByteRange> valid_ranges;
426 for (auto range : range_header->ranges) {
427 if (range.normalize(total_size)) {
428 valid_ranges.push_back(range);
429 }
430 }
431
432 if (valid_ranges.empty()) {
433 return range_not_satisfiable(total_size);
434 }
435
436 // Single range vs multipart
437 if (valid_ranges.size() == 1) {
438 return build_single_range(valid_ranges[0], total_size);
439 } else {
440 return build_multipart_range(valid_ranges, total_size);
441 }
442}
