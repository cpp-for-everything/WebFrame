99CookieJar CookieJar::parse(std::string_view header) {
100 CookieJar jar;
101
102 size_t start = 0;
103 while (start < header.size()) {
104 // Find end of this cookie (semicolon or end)
105 size_t end = header.find(';', start);
106 if (end == std::string_view::npos) {
107 end = header.size();
108 }
109
110 auto pair = trim(header.substr(start, end - start));
111
112 // Find = separator
113 auto eq = pair.find('=');
114 if (eq != std::string_view::npos) {
115 auto name = trim(pair.substr(0, eq));
116 auto value = trim(pair.substr(eq + 1));
117
118 // Remove quotes if present
119 if (value.size() >= 2 && value.front() == '"' && value.back() == '"') {
120 value = value.substr(1, value.size() - 2);
121 }
122
123 if (!name.empty()) {
124 jar.cookies_[std::string(name)] = std::string(value);
125 }
126 }
127
128 start = end + 1;
129 }
130
131 return jar;
132}
