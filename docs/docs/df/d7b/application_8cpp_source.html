200  int application::responder(SOCKET socket) {
201  request r;
202  try {
203  const std::size_t capacity = 1024;
204  char data[capacity];
205  int total_recv = 0;
206  int n = 0;
207  do {
208  n = RECV(socket, data, capacity, 0);
209  if (n < 0) {
210  break;
211  }
212 
213  total_recv += n;
214 
215  auto state = r.loadMore(data, n);
216 
217  if (state == LoadingState::LOADED) // Headers are loaded, the buffer
218  // is not full and the last char
219  // is \0 -> body is filled
220  break;
221  } while (n > 0);
222 
223  this->logger << "(responder) Read state: " << (int)r.getState() << " " << total_recv << "\n";
224 
225  if (r.getState() != LoadingState::LOADED)
226  throw std::string(
227  "Request was not loaded completely and data with "
228  "size=" +
229  std::to_string(total_recv) + " was sent.");
230 
231  const auto t1 = std::chrono::system_clock::now();
232  response res;
233  res = this->respond(r, r.http);
234  const std::string& response = res.to_string();
235  const size_t responseSize = response.size();
236  int status;
237  status = SEND(socket, response.c_str(), responseSize, 0);
238  if (status == SOCKET_ERROR) {
239  return -1;
240  }
241  this->performancer << r.uri << ": " << timer(t1).count() << "miliseconds\n";
242  } catch (std::exception const& e) {
243  this->errors << "(responder) Responding Exception: " << e.what() << "\n";
244  if (!this->routes.match("500").has_value()) throw std::invalid_argument("500 pge not found.");
245  response res =
246  this->routes.match("500").value().executor.call(r.http, "", {{std::string(e.what()), "string"}});
247  const std::string& response = res.to_string();
248  const size_t responseSize = response.size();
249  SEND(socket, response.c_str(), responseSize, 0);
250  return -1;
251  } catch (...) {
252  return -2;
253  }
254  return 0;
255  }

286  int status;
287  struct addrinfo hints, *res;
288  SOCKET listener;
289 
290  // Before using hint you have to make sure that the data
291  // structure is empty
292  memset(&hints, 0, sizeof hints);
293  // Set the attribute for hint
294  hints.ai_family = AF_INET; // We don't care V4 AF_INET or 6 AF_INET6
295  hints.ai_socktype = SOCK_STREAM; // TCP Socket SOCK_DGRAM
296  hints.ai_flags = AI_PASSIVE;
297  hints.ai_protocol = 0; /* Any protocol */
298  hints.ai_canonname = NULL;
299  hints.ai_addr = NULL;
300  hints.ai_next = NULL;
301 
302  // Fill the res data structure and make sure that the
303  // results make sense.
304  status = getaddrinfo(NULL, PORT, &hints, &res);
305  if (status != 0) {
306  this->errors << "(main) getaddrinfo error: " << gai_strerror(status) << "\n";
307  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
308  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
309  throw exceptions::unable_to_retrieve_address();
310  }
311 
312  // Create Socket and check if error occured afterwards
313  listener = socket(res->ai_family, res->ai_socktype, res->ai_protocol);
314  if (listener == INVALID_SOCKET) {
315  this->errors << "(main) socket error: " << gai_strerror(status) << "\n";
316  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
317  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
318  throw exceptions::socket_not_created();
319  }
320 
321  // Bind the socket to the address of my local machine and
322  // port number
323  status = bind(listener, res->ai_addr, sizeof(*res->ai_addr) /*res->ai_addrlen*/);
324  if (status < 0) {
325  this->errors << "(main) bind error: " << status << " " << gai_strerror(status) << "\n";
326  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
327  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
328  throw exceptions::unable_to_bind_socket();
329  }
330 
331  status = listen(listener, 128);
332  if (status < 0) {
333  this->errors << "(main) listen error: " << gai_strerror(status) << "\n";
334  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
335  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
336  throw exceptions::unable_to_bind_socket();
337  }
338 
339  // status = nonblock_config(listener);
340  status = nonblock_config(listener);
341  if (status < 0) {
342  this->errors << "(main) nonblocking config error: " << gai_strerror(status) << "\n";
343  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
344  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
345  throw exceptions::socket_not_created();
346  }
347 
348  // Free the res linked list after we are done with it

351  return listener;

355  std::function<void()> on_end) {
356  while (application::port_status.get_status(PORT.c_str()) !=
357  webframe::utils::server_status::port::Status::SIGNALED) {

362  }

366

381  << " was skipped (" << status << ")\n";

383  }

386  << " vs invalid socket " << INVALID_SOCKET << "\n";
