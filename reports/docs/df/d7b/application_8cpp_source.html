231  int application::responder(SOCKET socket) {
232  request r;
233  try {
234  const std::size_t capacity = 1024;
235  char data[capacity];
236  int total_recv = 0;
237  int n = 0;
238  do {
239  n = RECV(socket, data, capacity, 0);
240  if (n < 0) {
241  break;
242  }
243 
244  total_recv += n;
245 
246  auto state = r.loadMore(data, n);
247 
248  if (state ==
249  LoadingState::LOADED) // Headers are loaded, the buffer
250  // is not full and the last char
251  // is \0 -> body is filled
252  break;
253  } while (n > 0);
254 
255  this->logger << "(responder) Read state: " <<
256  (int)r.getState() << " "
257  << total_recv << "\n";
258 
259  if (r.getState() != LoadingState::LOADED)
260  throw std::string(
261  "Request was not loaded completely and data with "
262  "size=" +
263  std::to_string(total_recv) + " was sent.");
264 
265  const auto t1 = std::chrono::system_clock::now();
266  response res;
267  res = this->respond(r, r.http);
268  const std::string& response = res.to_string();
269  const size_t responseSize = response.size();
270  int status;
271  status = SEND(socket, response.c_str(), responseSize, 0);
272  if (status == SOCKET_ERROR) {
273  return -1;
274  }
275  this->performancer << r.uri << ": " << timer(t1).count()
276  << "miliseconds\n";
277  } catch (std::exception const& e) {
278  this->errors << "(responder) Responding Exception: " << e.what()
279  << "\n";
280  if (!this->routes.match("500").has_value())
281  throw std::invalid_argument("500 pge not found.");
282  response res = this->routes.match("500").value().executor.call(
283  r.http, "",
284  {{std::string(e.what()), "string"}});
285  const std::string& response = res.to_string();
286  const size_t responseSize = response.size();
287  SEND(socket, response.c_str(), responseSize, 0);
288  return -1;
289  } catch (...) {
290  return -2;
291  }
292  return 0;

324  int status;
325  struct addrinfo hints, *res;
326  SOCKET listener;
327 
328  // Before using hint you have to make sure that the data
329  // structure is empty
330  memset(&hints, 0, sizeof hints);
331  // Set the attribute for hint
332  hints.ai_family = AF_INET; // We don't care V4 AF_INET or 6 AF_INET6
333  hints.ai_socktype = SOCK_STREAM; // TCP Socket SOCK_DGRAM
334  hints.ai_flags = AI_PASSIVE;
335  hints.ai_protocol = 0; /* Any protocol */
336  hints.ai_canonname = NULL;
337  hints.ai_addr = NULL;
338  hints.ai_next = NULL;
339 
340  // Fill the res data structure and make sure that the
341  // results make sense.
342  status = getaddrinfo(NULL, PORT, &hints, &res);
343  if (status != 0) {
344  this->errors << "(main) getaddrinfo error: " << gai_strerror(status) << "\n";
345  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
346  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
347  throw exceptions::unable_to_retrieve_address();
348  }
349 
350  // Create Socket and check if error occured afterwards
351  listener = socket(res->ai_family, res->ai_socktype,
352  res->ai_protocol);
353  if (listener == INVALID_SOCKET) {
354  this->errors
355  << "(main) socket error: " << gai_strerror(status)
356  << "\n";
357  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STARTED);
358  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
359  throw exceptions::socket_not_created();
360  }
361

370  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
371  throw exceptions::unable_to_bind_socket();
372  }
373

375  if (status < 0) {
376  this->errors

380  application::port_status.change_status(PORT, webframe::utils::server_status::port::Status::STOPPED);
381  throw exceptions::unable_to_bind_socket();
